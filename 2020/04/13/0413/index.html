case

多重筛选

  1. 语法结构
    ps: pattern 需要匹配的模式

case var in
pattern1) 模式1;用|分割多个模式，相当于or
command1
;; 两个分号代表命令结束
pattern2)
command2
;;
*)
command3
;;
esac

2. 案例
> 输入不同的传参执行不同的命令
```bash 
#!/bin/bash
case $1 in
name|N)
    echo "my name is lakesayriver"
    ;;
favorite|F)
    echo "I like watching movies"
    ;;
*)
    echo "hello! I really like you so much"
    ;;
esac

根据用户的需求执行命令

tips: set list 可以查看脚本的格式，比如制表符，空格等

需求如下

h    显示命令帮助
f    显示磁盘分区
d    显示磁盘挂载
m    查看内存使用
u    查询系统负载
q    退出程序

#!/bin/bash
#打印菜单,'<<-'表示后面的END不用靠墙
cat <<-END
    h    显示命令帮助
    f    显示磁盘分区
    d    显示磁盘挂载
    m    查看内存使用
    u    查询系统负载
    q    退出程序
    END
while true
do
#用户选择菜单
read -p "请选择要操作的菜单（help h）:" action
case $action in
h|help)
    cat <<-END
        h    显示命令帮助
        f    显示磁盘分区
        d    显示磁盘挂载
        m    查看内存使用
        u    查询系统负载
        q    退出程序
        END
    ;;
f)
    lsblk
    ;;
d)
    df -h
    ;;
m)
    free -m
    ;;
u)
    uptime
    ;;
q)
    exit
    ;;
esac
done
